108class LeeDistance final : public Distance<LeeDistance> {
109 integer_type m_alphabet_size;
110
111public:
112 explicit LeeDistance(integer_type alphabet_size = BK_LEE_ALPHABET_SIZE)
113 : m_alphabet_size(alphabet_size){};
114 integer_type compute_distance(std::string_view s, std::string_view t) const {
115 const integer_type M = s.length(), N = t.length();
116 if (M != N) {
117 return std::numeric_limits<integer_type>::max();
118 }
119 const integer_type m_comparsion_size = M;
120 integer_type counter = 0, diff;
121 for (integer_type i = 0; i < m_comparsion_size; ++i) {
122 diff = std::abs(s[i] - t[i]);
123 counter += std::min(diff, m_alphabet_size - diff);
124 }
125 return counter;
126 }
127};

144class LCSubseqDistance final : public Distance<LCSubseqDistance> {
145 mutable std::vector<integer_type> m_current, m_previous;
146
147public:
148 explicit LCSubseqDistance(size_t initial_size = BK_LCS_MATRIX_INITIAL_SIZE)
149 : m_current(initial_size), m_previous(initial_size){};
150 integer_type compute_distance(std::string_view s, std::string_view t) const {
151 const integer_type M = s.length(), N = t.length();
152 if (M == 0 || N == 0) {
153 return 0;
154 }
155 if (m_current.size() <= N || m_previous.size() <= N) {
156 m_current.resize(N + 1);
157 m_previous.resize(N + 1);
158 }
159 std::fill(m_previous.begin(), m_previous.end(), 0);
160 for (integer_type i = 1; i <= M; ++i) {
161 for (integer_type j = 1; j <= N; ++j) {
162 if (s[i - 1] == t[j - 1]) {
163 m_current[j] = m_previous[j - 1] + 1;
164 } else {
165 m_current[j] = std::max(m_previous[j], m_current[j - 1]);
166 }
167 }
168 m_previous = m_current;
169 }
170 return m_previous[N];
171 }

222class EditDistance final : public Distance<EditDistance> {
223 mutable std::vector<std::vector<integer_type>> m_matrix;
224
225public:
226 explicit EditDistance(size_t initial_size = BK_ED_MATRIX_INITIAL_SIZE)
227 : m_matrix(initial_size, std::vector<integer_type>(initial_size)){};
228 integer_type compute_distance(std::string_view s, std::string_view t) const {
229 const integer_type M = s.length(), N = t.length();
230 if (M == 0 || N == 0) {
231 return N + M;
232 }
233 if (m_matrix.size() <= M || m_matrix[0].size() <= N) {
234 std::vector<std::vector<integer_type>> a_matrix(M + 1,
235 std::vector<integer_type>(N + 1));
236 m_matrix.swap(a_matrix);
237 }
238 for (integer_type i = 1; i <= M; ++i) {
239 m_matrix[i][0] = i;
240 }
241 for (integer_type j = 1; j <= N; ++j) {
242 m_matrix[0][j] = j;
243 }
244 for (integer_type j = 1; j <= N; ++j) {
245 for (integer_type i = 1; i <= M; ++i) {
246 m_matrix[i][j] = std::min(
247 {m_matrix[i][j - 1] + 1 /*Insertion*/, m_matrix[i - 1][j] + 1 /*Deletion*/,
248 m_matrix[i - 1][j - 1] + (s[i - 1] != t[j - 1]) /*Substitution*/});
249 }
250 }
251 return m_matrix[M][N];
252 }

260class DamerauLevenshteinDistance final : public Distance<DamerauLevenshteinDistance> {
261 mutable std::vector<std::vector<integer_type>> m_matrix;
262
263public:
264 explicit DamerauLevenshteinDistance(size_t initial_size = BK_MATRIX_INITIAL_SIZE)
265 : m_matrix(initial_size, std::vector<integer_type>(initial_size)){};
266 integer_type compute_distance(std::string_view s, std::string_view t) const {
267 const integer_type M = s.length(), N = t.length();
268 if (M == 0 || N == 0) {
269 return N + M;
270 }
271 if (m_matrix.size() <= M || m_matrix[0].size() <= N) {
272 std::vector<std::vector<integer_type>> a_matrix(M + 1,
273 std::vector<integer_type>(N + 1));
274 m_matrix.swap(a_matrix);
275 }
276 for (integer_type i = 1; i <= M; ++i) {
277 m_matrix[i][0] = i;
278 }
279 for (integer_type j = 1; j <= N; ++j) {
280 m_matrix[0][j] = j;
281 }
282 for (integer_type j = 1; j <= N; ++j) {
283 for (integer_type i = 1; i <= M; ++i) {
284 m_matrix[i][j] = std::min(
285 {m_matrix[i][j - 1] + 1 /*Insertion*/, m_matrix[i - 1][j] + 1 /*Deletion*/,
286 m_matrix[i - 1][j - 1] + (s[i - 1] == t[j - 1] ? 0 : 1) /*Substitution*/});
287 if (i > 1 && j > 1 && s[i - 1] == t[j - 2] && s[i - 2] == t[j - 1]) {
288 m_matrix[i][j] =
289 std::min(m_matrix[i][j], m_matrix[i - 2][j - 2] + 1 /*Transposition*/);
290 }
291 }
292 }
293 return m_matrix[M][N];
294 }

371 reference operator*() const { return *m_pointer; }
372
373 Iterator &operator++() {
374 if (m_pointer == nullptr) {
375 throw std::out_of_range("No more tree node");
376 }
377 for (auto &[_, child] : (*m_pointer)->m_children) {
378 m_queue.push(&child);
379 }
380 if (m_queue.empty()) {
381 m_pointer = nullptr;
382 } else {
383 m_pointer = m_queue.front();
384 m_queue.pop();
385 }
386 return *this;
387 }

390 Iterator tmp{*this};
391 ++(*this);
392 return tmp;
393 }

395 friend bool operator==(const Iterator &a, const Iterator &b) {
396 return a.m_pointer == b.m_pointer;
397 };
398
399 friend bool operator!=(const Iterator &a, const Iterator &b) {
400 return a.m_pointer != b.m_pointer;
401 };

403 private:
404 pointer m_pointer;
405 std::queue<pointer> m_queue;
406 };

419 BKTree(const BKTree &other) : BKTree(other.m_metric) {
420 if (other.m_root == nullptr) {

423 std::queue<std::unique_ptr<node_type> const *> bq;
424 bq.push(&(other.m_root));
425 while (!bq.empty()) {
426 auto *nptr = bq.front();
427 bq.pop();
428 this->insert((*nptr)->m_word);
429 for (auto &[_, child_node] : (*nptr)->m_children) {
430 bq.push(&child_node);
431 }
432 }

471template <typename Metric>
472bool BKTreeNode<Metric>::_insert(std::string_view value,
473 const metric_type &distance_metric) {
474 const int distance_between = distance_metric(value, m_word);
475 bool inserted = false;
476 if (distance_between >= 0) {
477 auto it = m_children.find(distance_between);
478 if (it == m_children.end()) {
479 m_children.emplace(std::make_pair(
480 distance_between, std::unique_ptr<node_type>(new node_type(value))));
481 inserted = true;
482 } else {
483 inserted = it->second->_insert(value, distance_metric);
484 }
485 }
486 return inserted;
487}

489template <typename Metric>
490bool BKTreeNode<Metric>::_erase(std::string_view value,
491 const metric_type &distance_metric) {
492 bool erased = false;
493 const int distance_between = distance_metric(value, m_word);
494 auto it = m_children.find(distance_between);
495 if (it != m_children.end()) {
496 if (it->second->m_word == value) {
497 auto node = std::move(it->second);
498 m_children.erase(it);
499 std::queue<std::unique_ptr<node_type> const *> bq;
500 for (auto const &[_, child_node] : node->m_children) {
501 bq.push(&child_node);
502 }
503 while (!bq.empty()) {
504 auto *node = bq.front();
505 bq.pop();
506 for (auto const &[_, child_node] : (*node)->m_children) {
507 bq.push(&child_node);
508 }
509 _insert((*node)->m_word, distance_metric);
510 }
511 erased = true;
512 } else {
513 erased = it->second->_erase(value, distance_metric);
514 }
515 } else {
516 for (auto const &[_, child] : m_children) {
517 if (child->_erase(value, distance_metric)) {
518 return true;
519 }
520 }
521 }
522 return erased;
523}

525template <typename Metric>
526void BKTreeNode<Metric>::_find(ResultList &output, std::string_view value,
527 const int &limit, const metric_type &metric) const {
528 const int distance = metric(value, m_word);
529 if (distance <= limit) {
530 output.push_back({m_word, distance});
531 }
532 for (auto const &[dist, node] : m_children) {
533 if (std::abs(dist - distance) <= limit) {
534 node->_find(output, value, limit, metric);
535 }
536 }
537}

548bool BKTree<Metric>::insert(std::string_view value) {

552 ++m_tree_size;

561template <typename Metric>
562bool BKTree<Metric>::erase(std::string_view value) {
563 bool erased = false;
564 if (m_root == nullptr) {
565 erased = true;
566 } else if (m_root->m_word == value) {
567 if (m_tree_size > 1) {
568 auto &replacement_node = m_root->m_children.begin()->second;
569 std::queue<std::unique_ptr<node_type> const *> bq;
570 for (bool first = true; auto const &[_, node] : m_root->m_children) {
571 if (first) {
572 first = false;
573 continue;
574 }
575 bq.push(&node);
576 }
577 while (!bq.empty()) {
578 auto node = bq.front();
579 bq.pop();
580 for (auto const &[_, child] : (*node)->m_children) {
581 bq.push(&child);
582 }
583 replacement_node->_insert((*node)->m_word, m_metric);
584 }
585 m_root = std::move(replacement_node);
586 } else {
587 m_root.reset(nullptr);
588 }
589 --m_tree_size;
590 erased = true;
591 } else if (m_root->_erase(value, m_metric)) {
592 --m_tree_size;
593 erased = true;
594 }
595 return erased;
596}
